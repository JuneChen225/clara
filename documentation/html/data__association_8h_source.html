95  const std::vector<cone_state<T>> &
96  classify_new_data(const std::vector<raw_cone_data> & new_cones)
97  {
98  _detected_cluster_ix.clear();
99  // when we start the car, we will take every visible cone as a new cluster, we have the assumption that we don't see cones twice in a single image
100  if (_cone_states.size() == 0)
101  {
102  for(auto cone : new_cones)
103  {
104  _add_new_cluster_with_ob(cone);
105  }
106  }
107  // otherwise, we need to detect new cones and associate current clusters to our observations
108  else
109  {
110  for(auto cone : new_cones){
111  auto cluster_it = _get_most_probable_cluster_it(cone);
112  if ((*cluster_it).distance_greater_than(cone, _max_dist_btw_cones_m))
113  {
114  _add_new_cluster_with_ob(cone);
115  }
116  // if it's not below the threshold, we need to put it in an existing cluster
117  else
118  {
119  _add_observation(cluster_it, cone);
120  }
121  }
122  }
123  return _cone_states;
124  }

149  const std::vector<cone_state<T>> get_detected_cluster_sorted(const std::tuple<double, double> car_pos) const
150  {
151  std::vector<cone_state<T>> detected_cluster = get_detected_cluster();
152  std::sort(detected_cluster.begin(), detected_cluster.end(), [&](const cone_state<T> & a, const cone_state<T> & b)
153  {
154  const double car_x = std::get<0>(car_pos);
155  const double car_y = std::get<1>(car_pos);
156  const double dist_A = a.distance(car_x, car_y);
157  const double dist_B = b.distance(car_x, car_y);
158  return dist_A < dist_B;
159  });
160  return detected_cluster;
161  }
