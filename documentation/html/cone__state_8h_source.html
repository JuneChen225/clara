153  double pdf(T x, T y)
154  {
155  update_state();
156 
157  const double fraction = 1 / std::sqrt(std::pow(2*M_PI, 2) * _det_cov_mat);
158 
159  const T xx = _inv_cov_mat[0];
160  const T xy = _inv_cov_mat[1];
161  const T yy = _inv_cov_mat[3];
162  const T x_mean = _mean_vec[0];
163  const T y_mean = _mean_vec[1];
164  const T x_norm = x - x_mean;
165  const T y_norm = y - y_mean;
166 
167  const double mahalanobis = 2*(x_norm*y_norm*xy) + std::pow(x_norm,2) * xx + std::pow(y_norm,2) * yy;
168  const double factor = std::exp(-0.5 * mahalanobis);
169 
170  return fraction * factor;
171  }

267  void update_mean_vec()
268  {
269  const coords_t sums =
270  std::accumulate(_observations.begin(), _observations.end()
271  , coords_t { { 0, 0 } }
272  , [](coords_t acc, coords_t _o)
273  {
274  const T x = _o[0];
275  const T y = _o[1];
276  acc[0] += x;
277  acc[1] += y;
278  return acc;
279  });
280 
281  const T x_sum = sums[0];
282  const T y_sum = sums[1];
283  const size_t observations_count = _observations.size();
284 
285  _mean_vec[0] = x_sum / observations_count;
286  _mean_vec[1] = y_sum / observations_count;
287  }

294  void update_cov_mat()
295  {
296  const T mean_x = _mean_vec[0];
297  const T mean_y = _mean_vec[1];
298  const size_t observations_count = _observations.size();
299 
300  const std::array<T, 3> cov_sum =
301  std::accumulate(_observations.begin(), _observations.end()
302  , std::array<T,3> { { 0, 0, 0 } }
303  , [&](std::array<T, 3> & cov_sum, const coords_t & _o)
304  {
305  const T x = _o[0];
306  const T y = _o[1];
307  const T x_factor = x - mean_x;
308  const T y_factor = y - mean_y;
309  const T xx_summand = std::pow(x_factor, 2);
310  const T yy_summand = std::pow(y_factor, 2);
311  const T xy_summand = x_factor * y_factor;
312 
313  cov_sum[0] += xx_summand;
314  cov_sum[1] += yy_summand;
315  cov_sum[2] += xy_summand;
316  return cov_sum;
317  });
318 
319  const T cov_xx = cov_sum[0] / observations_count;
320  const T cov_yy = cov_sum[1] / observations_count;
321  const T cov_xy = cov_sum[2] / observations_count;
322 
323  _cov_mat[0] = cov_xx;
324  _cov_mat[1] = cov_xy;
325  _cov_mat[2] = cov_xy;
326  _cov_mat[3] = cov_yy;
327  }
