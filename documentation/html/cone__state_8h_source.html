120  double maximum_likelihood_estimate(T x, T y)
121  {
122  const double fraction = 1 / std::sqrt(2*std::pow(M_PI, 2) * _det_cov_mat);
123 
124  const T xx = _inv_cov_mat[0];
125  const T xy = _inv_cov_mat[1];
126  const T yy = _inv_cov_mat[3];
127  const double mahalanobis = 2*(x*y*xy) + std::pow(x,2) * xx + std::pow(y,2) * yy;
128  const double factor = std::exp(-0.5 * mahalanobis);
129 
130  std::cout << "\nmle:\n"

132  << " factor: " << factor << '\n'
133  << " fraction: " << fraction << '\n';
134 
135  return fraction * factor;

245  void update_cov_mat()
246  {
247  const T mean_x = _mean_vec[0];
248  const T mean_y = _mean_vec[1];
249  const size_t observations_count = _observations.size();
250 
251  const std::array<T, 3> cov_sum =
252  std::accumulate(_observations.begin(), _observations.end()
253  , std::array<T,3> { 0, 0, 0 }
254  , [&](std::array<T, 3> & cov_sum, const coords_t & _o)
255  {
256  const T x = _o[0];
257  const T y = _o[1];
258  const T x_factor = x - mean_x;
259  const T y_factor = y - mean_y;
260  const T xx_summand = std::pow(x_factor, 2);
261  const T yy_summand = std::pow(y_factor, 2);
262  const T xy_summand = x_factor * y_factor;
263 
264  cov_sum[0] += xx_summand;
265  cov_sum[1] += yy_summand;
266  cov_sum[2] += xy_summand;
267  return cov_sum;
268  });
269 
270  const T cov_xx = cov_sum[0] * 1 / observations_count;
271  const T cov_yy = cov_sum[1] * 1 / observations_count;
272  const T cov_xy = cov_sum[2] * 1 / observations_count;
273 
274  _cov_mat[0] = cov_xx;
275  _cov_mat[1] = cov_xy;
276  _cov_mat[2] = cov_xy;
277  _cov_mat[3] = cov_yy;
278

280  << "_cov_mat[0]: " << _cov_mat[0] << '\n'

282  << "_cov_mat[2]: " << _cov_mat[2] << '\n'
